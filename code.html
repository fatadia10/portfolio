🔐 SecureAccess Manager

🐙 Voir sur GitHub

Application console en langage C développée à l'UNCHK. Gestion d'utilisateurs avec niveaux d'accès, vérification de mots de passe et persistance en fichier binaire.

/* SecureAccess Manager — Bineta DIA — UNCHK 2024/2025 */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

/* Structure représentant un utilisateur */
typedef struct {
    int id;
    char username[50];
    char password[50];
    int access_level; /* 1: standard, 2: admin */
} User;

/* Vide le buffer stdin */
void clean_stdin() {
    int c;
    while ((c = getchar()) != '\n' && c != EOF);
}

/* Ajoute un nouvel utilisateur */
void addUser(User **users, int *count, int *nextId) {
    User *temp = (User *)realloc(*users, (*count + 1) * sizeof(User));
    if (temp == NULL) {
        printf("Erreur : Mémoire insuffisante.\n");
        return;
    }
    *users = temp;
    User newUser;
    newUser.id = (*nextId)++;
    printf("\n--- Ajouter un utilisateur ---\n");
    printf("Nom d'utilisateur : ");
    scanf("%49s", newUser.username);
    printf("Mot de passe : ");
    scanf("%49s", newUser.password);
    printf("Niveau d'accès (1: standard, 2: admin) : ");
    while(scanf("%d", &newUser.access_level) != 1) {
        printf("Entrée invalide. Saisissez un nombre : ");
        clean_stdin();
    }
    (*users)[*count] = newUser;
    (*count)++;
    printf("Utilisateur ajouté (ID: %d).\n", newUser.id);
}

/* Vérifie la force d'un mot de passe */
int checkPasswordStrength(const char *password) {
    int hasDigit = 0;
    if (strlen(password) < 6) return 0;
    for (int i = 0; password[i] != '\0'; i++) {
        if (isdigit(password[i])) hasDigit = 1;
    }
    return hasDigit;
}

/* Sauvegarde les utilisateurs dans un fichier binaire */
void saveUsers(User *users, int count, const char *filename) {
    FILE *file = fopen(filename, "wb");
    if (file == NULL) { perror("Erreur sauvegarde"); return; }
    fwrite(&count, sizeof(int), 1, file);
    fwrite(users, sizeof(User), count, file);
    fclose(file);
    printf("Données sauvegardées dans %s.\n", filename);
}

int main() {
    User *users = NULL;
    int count = 0, nextId = 1, choix;
    const char *filename = "users.dat";

    printf("=== SecureAccess Manager ===\n");
    do {
        printf("\n1. Ajouter\n2. Afficher\n3. Supprimer\n4. Sauvegarder\n5. Vérifier\n0. Quitter\n> ");
        scanf("%d", &choix);
        clean_stdin();
        switch(choix) {
            case 1: addUser(&users, &count, &nextId); break;
            case 4: saveUsers(users, count, filename); break;
            case 0: printf("Au revoir !\n"); break;
            default: printf("Option invalide.\n");
        }
    } while(choix != 0);
    free(users);
    return 0;
}
